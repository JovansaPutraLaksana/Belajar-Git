#Repository Local/Hanya pada komputer pribadi (Belum Remote ke Repository Github Public)

#The Three States (Modified,Staged,Committed)
#Modified = file telah berubah namun belum disimpan secara permanen ke repository
#Staged = menandai modifikasi yang telah dilakukan terhadap file akan disimpan ke repository
#Committed = file sudah tersimpan di repository

#Three Section (Working directory, Staging area, Repository)
#Working directory = tempat file akan dimodifikasi
#Staging area = tempat persiapan file akan di simpan di repository
#Repository = tempat semua file riwayat snapshot(kumpulan perubahan) disimpan

#Tahap workflow Git
#Working directory -> (git add) -> Staging area -> (git commit) -> Repository 

#Perintah dasar cmd
cd <nama folder> (masuk kedalam directory)
cd .. (mundur dari directory)
mkdir <nama folder> (membuat folder baru pada directory)
dir (melihat file normal dari folder)
dir /a (melihat semua isi dari folder)
dir /a:h (melihat file hidden dari folder)

#Cara membuat repository Git pada directory
git init

#Melihat perubahan pada git (Merah = Working directory, Hijau = Staging area)
git status

#Memindahkan file dari Working directory ke Staging area.
git add .(untuk semua file)
git add <nama file> (khusus yang yang disebut (case sensitive))

#Memindahkan file dari Staging area ke Repository
git commit -m "<pesan keterangan>"

#Melihat perubahan yan terjadi pada file
git diff
git diff <hash> HEAD (membandingkan perubahan dari snapshot hash ke HEAD (snapshot sekarang))
git difftool <hash1> <hash2> (membandingkan di VSCode wajib setting konfigurasi)

#Hapus perubahan di Working directory
git clean -f

#Membatalkan perubahan file di Working directory
git restore <nama file>

#Memindahkan file dari Staging area ke Working directory
git restore --staged <nama file>

#Melihat commit log (history snapshot) hash didapatkan dari log warna kuning
git log
git log --oneline (sederhana)
git log --oneline --graph (melihat graph dari git branch)
git show <hash>

#Reset commit (mundur HEAD pada commit ke hash yang diinginkan)
#mode reset => --soft (Staging area dan Working directory tidak berubah/dihapus)
#mode reset => --mixed (Staging area reset tetapi Working directory aman)
#mode reset => --hard (Reset semua Stage kembali saat commit HEAD sebelumnya)
#Selagi belum commit baru. maka bisa batalkan reset commit apabila masih ingat hash dari HEAD sebelumnya 
git reset <mode> <hash yang akan menjadi HEAD>

#Amend commit (commit ketingalan)
git commit --amend -m "<pesan>"

#Melihat isi file dari commit sebelumnya dengan cara mengambil isi file dan disimpan ke Staging area
git checkout <hash> -- <nama file>
git checkout <hash> (semua file)
git checkout <nama branch> (kembali ke HEAD dari branch)

#Melihat posisi branch sekarang
git branch --show-current
